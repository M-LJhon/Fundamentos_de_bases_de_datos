-- Preparar datos de ejemplo
CREATE TABLE IF NOT EXISTS paises (
    pais_id INTEGER PRIMARY KEY,
    nombre  TEXT NOT NULL,
    codigo  VARCHAR(3) NOT NULL
);

CREATE TABLE IF NOT EXISTS clientes (
    cliente_id INTEGER PRIMARY KEY,
    nombre     TEXT NOT NULL,
    email      VARCHAR(100),
    activo     BOOLEAN,
    telefono   VARCHAR(20)
);

CREATE TABLE IF NOT EXISTS productos (
    producto_id INTEGER PRIMARY KEY,
    nombre      TEXT NOT NULL,
    precio      NUMERIC(12,2),
    stock       INTEGER
);

-- SELECT -> Seleccionamos todas las columnas de clientes
-- Pero limitamos a 2 registros o filas.
SELECT * FROM clientes LIMIT 2;

-- SELECT -> Especificando los nombres de las columnas.
SELECT
    nombre, email, activo
FROM clientes LIMIT 10;

-- Filtrar datos.
-- WHERE: Filtrar por condicion simple.
SELECT nombre, email
FROM clientes
WHERE activo = FALSE;

SELECT nombre, precio
FROM productos
WHERE precio > 100;

-- WHERE con rango: BETWEEN 50 AND 200;
SELECT nombre, precio
FROM productos
WHERE precio BETWEEN 50 AND 200;

-- WHERE con listas: IN();
SELECT nombre, codigo
FROM paises
WHERE codigo IN ('CHL', 'ARG', 'COL');

-- WHERE con LIKE.
SELECT nombre, email
FROM clientes
WHERE nombre LIKE 'Mar%';

SELECT nombre, email
FROM clientes
WHERE email LIKE '%@gmail.com';

-- IS NULL / IS NOT NULL
SELECT nombre, telefono
FROM clientes
WHERE telefono IS NOT NULL;

-- AND / OR
SELECT nombre, precio, stock
FROM productos
WHERE precio > 50 AND stock > 0;

SELECT nombre, codigo
FROM paises
WHERE codigo = 'CHL' OR codigo = 'ARG';

-- NOT
SELECT nombre, activo
FROM clientes
WHERE not activo = FALSE;

-- ORDER BY
SELECT nombre, precio
FROM productos
ORDER BY precio ASC
LIMIT 10;

SELECT nombre, precio
FROM productos
ORDER BY precio DESC
LIMIT 10;

-- Alias con AS
SELECT nombre AS cliente,
       email  AS correo,
       activo AS esta_activo
FROM clientes
LIMIT 5;

-- Limpieza
DROP TABLE IF EXISTS clientes;
DROP TABLE IF EXISTS productos;
DROP TABLE IF EXISTS paises;
